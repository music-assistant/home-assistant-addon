#include <tunables/global>

profile music_assistant_dev flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/dbus-strict>

  capability,
  file,
  signal,
  mount,
  umount,
  remount,
  network udp,
  network tcp,
  network dgram,
  network stream,
  network seqpacket,
  network inet,
  network inet6,
  network netlink raw,
  network unix dgram,

  capability setgid,
  capability setuid,
  capability sys_admin,
  capability dac_read_search,

  /dev/* mrwkl,
  /tmp/** mrkwl,

  # Data access
  /data/** rw,
  /media/** rw,

  # D-Bus system bus access (required for shairport-sync remote control)
  dbus bus=system,
  unix (send, receive) type=stream addr=none peer=(label=unconfined),
  /run/dbus/system_bus_socket rw,

  # docker daemon confinement requires explict allow rule for signal
  signal (receive) set=(kill,term) peer=/usr/bin/docker,

}
